export function useBlockchainQuery<T>(
  query: string,
  params?: any[]
) {
  const { data, isLoading } = useQuery({
    queryKey: ['blockchain', query, params],
    queryFn: () => api.blockchain.query(query, params),
  });

}

export function useBlockchainTx() {
  const mutation = useMutation({
    mutationFn: (tx: BlockchainTx) => 
      api.blockchain.sendTransaction(tx),
  });
  
  return { sendTx: mutation.mutate };
}
